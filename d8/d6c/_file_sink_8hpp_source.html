359  void removeFilesIfNecessary()
360  {
361  // ToDo: use c++20 ranges::view
362  auto directoryItr = std::filesystem::directory_iterator(m_CurrentFilePath->parent_path());
363 
364  std::vector<std::filesystem::directory_entry> files;
365  for (const auto& entry : directoryItr)

386  void fulfillFileCountCleanup(std::vector<std::filesystem::directory_entry>& files) const
387  {
388  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
389  if (!cleanupRule.fileCount)

400  void fulfillDirectorySizeCleanup(std::vector<std::filesystem::directory_entry>& files) const
401  {
402  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
403  if (!cleanupRule.directorySize)
404  return;
405 
406  auto size = std::accumulate(
407  std::begin(files),
408  std::end(files),
409  0ull,
410  [](auto value, const auto& file) { return value + file.file_size(); }
411  );
412  while (*cleanupRule.directorySize < size)
413  {
414  auto& file = files.back();
415  size -= file_size(file);
416  remove(file);
417  files.pop_back();
418  }
