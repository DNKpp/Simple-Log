300  void removeFilesIfNecessary()
301  {
302  // ToDo: use c++20 ranges::view
303  auto directoryItr = std::filesystem::directory_iterator(m_CurrentFilePath->parent_path());
304 
305  std::vector<std::filesystem::directory_entry> files;
306  for (const auto& entry : directoryItr)
307  {
308  if (is_regular_file(entry) && entry.path().extension() == m_CurrentFilePath->extension() &&
309  !equivalent(entry.path(), *m_CurrentFilePath))
310  {
311  files.emplace_back(entry);
312  }
313  }
314 
315  std::ranges::sort(files, std::greater(), [](const auto& file) { return last_write_time(file); });
316  fulfillFileCountCleanup(files);
317  fulfillDirectorySizeCleanup(files);
318  }

327  void fulfillFileCountCleanup(std::vector<std::filesystem::directory_entry>& files) const
328  {
329  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
330  if (!cleanupRule.fileCount)
331  return;
332 
333  while (*cleanupRule.fileCount < std::size(files))
334  {
335  auto& file = files.back();
336  remove(file);
337  files.pop_back();
338  }
339  }

341  void fulfillDirectorySizeCleanup(std::vector<std::filesystem::directory_entry>& files) const
342  {
343  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
344  if (!cleanupRule.directorySize)
345  return;
346 
347  auto size = std::accumulate(
348  std::begin(files),
349  std::end(files),
350  0ull,
351  [](auto value, const auto& file) { return value + file.file_size(); }
352  );
353  while (*cleanupRule.directorySize < size)
354  {
355  auto& file = files.back();
356  size -= file_size(file);
357  remove(file);
358  files.pop_back();
359  }
360  }

367  auto rotationRule = load(m_RotationRule, m_RotationRuleMx);
368  return (rotationRule.fileSize && *rotationRule.fileSize < file_size(*m_CurrentFilePath)) ||
369  (rotationRule.duration && m_FileOpeningTime.load() + *rotationRule.duration < std::chrono::steady_clock::now());
