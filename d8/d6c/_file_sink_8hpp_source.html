292  void removeFilesIfNecessary()
293  {
294  // ToDo: use c++20 ranges::view
295  auto directoryItr = std::filesystem::directory_iterator(m_CurrentFilePath->parent_path());
296 
297  std::vector<std::filesystem::directory_entry> files;
298  for (const auto& entry : directoryItr)
299  {
300  if (is_regular_file(entry) && entry.path().extension() == m_CurrentFilePath->extension() &&
301  !equivalent(entry.path(), *m_CurrentFilePath))
302  {
303  files.emplace_back(entry);
304  }
305  }
306 
307  std::ranges::sort(files, std::greater(), [](const auto& file) { return last_write_time(file); });
308  fulfillFileCountCleanup(files);
309  fulfillDirectorySizeCleanup(files);
310  }

319  void fulfillFileCountCleanup(std::vector<std::filesystem::directory_entry>& files) const
320  {
321  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
322  if (!cleanupRule.fileCount)
323  return;
324 
325  while (*cleanupRule.fileCount < std::size(files))
326  {
327  auto& file = files.back();
328  remove(file);
329  files.pop_back();
330  }
331  }

333  void fulfillDirectorySizeCleanup(std::vector<std::filesystem::directory_entry>& files) const
334  {
335  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
336  if (!cleanupRule.directorySize)
337  return;
338 
339  auto size = std::accumulate(
340  std::begin(files),
341  std::end(files),
342  0ull,
343  [](auto value, const auto& file) { return value + file.file_size(); }
344  );
345  while (*cleanupRule.directorySize < size)
346  {
347  auto& file = files.back();
348  size -= file_size(file);
349  remove(file);
350  files.pop_back();
351  }
352  }
