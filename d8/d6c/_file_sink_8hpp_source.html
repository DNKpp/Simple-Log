181  void removeFilesIfNecessary()
182  {
183  // ToDo: use c++20 ranges::view
184  auto directoryItr = std::filesystem::directory_iterator(m_CurrentFilePath->parent_path());
185 
186  std::vector<std::filesystem::directory_entry> files;
187  for (const auto& entry : directoryItr)
188  {
189  if (is_regular_file(entry) && entry.path().extension() == m_CurrentFilePath->extension() &&
190  !equivalent(entry.path(), *m_CurrentFilePath))
191  {
192  files.emplace_back(entry);
193  }
194  }
195 
196  std::ranges::sort(files, std::greater(), [](const auto& file) { return last_write_time(file); });
197  fulfillFileCountCleanup(files);
198  fulfillDirectorySizeCleanup(files);
199  }

208  void fulfillFileCountCleanup(std::vector<std::filesystem::directory_entry>& files) const
209  {
210  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
211  if (!cleanupRule.fileCount)
212  return;
213 
214  while (*cleanupRule.fileCount < std::size(files))
215  {
216  auto& file = files.back();
217  remove(file);
218  files.pop_back();
219  }
220  }
221 
222  void fulfillDirectorySizeCleanup(std::vector<std::filesystem::directory_entry>& files) const
223  {
224  auto cleanupRule = load(m_CleanupRule, m_CleanupRuleMx);
225  if (!cleanupRule.directorySize)
226  return;
227 
228  auto size = std::accumulate(
229  std::begin(files),
230  std::end(files),
231  0ull,
232  [](auto value, const auto& file) { return value + file.file_size(); }
233  );
234  while (*cleanupRule.directorySize < size)
235  {
236  auto& file = files.back();
237  size -= file_size(file);
238  remove(file);
239  files.pop_back();
240  }
241  }

243  [[nodiscard]] bool shallRotate() const
244  {
245  assert(m_FileStream.is_open() && m_CurrentFilePath && !std::empty(*m_CurrentFilePath));
246 
247  auto rotationRule = load(m_RotationRule, m_RotationRuleMx);
248  return (rotationRule.fileSize && *rotationRule.fileSize < file_size(*m_CurrentFilePath)) ||
249  (rotationRule.duration && m_FileOpeningTime.load() + *rotationRule.duration < std::chrono::steady_clock::now());
