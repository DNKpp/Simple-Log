330  inline std::ostream& applyTextColor(std::ostream& out, ConsoleTextStyle::Color color) noexcept
331  {
332  using Color = ConsoleTextStyle::Color;
333  if (color < Color::brightBlack)
334  {
335  const auto begin = static_cast<unsigned>(rang::fg::black);
336  out << static_cast<rang::fg>(static_cast<unsigned>(color) + begin);
337  }
338  else
339  {
340  const auto begin = static_cast<unsigned>(rang::fgB::black);
341  const auto localColor = static_cast<Color>(begin +
342  static_cast<unsigned>(color) -
343  static_cast<unsigned>(Color::brightBlack));
344  out << static_cast<rang::fgB>(localColor);
345  }
346  return out;
347  }
348
